帧动画播放 此方法主要针对需要兼容IE9以下浏览器

用来控制帧动画播放

HTML配置


<style>
.img1{width: 170px;height: 236px;display: block;margin:10px;background:url(a-lz.png) no-repeat;}
.img2{width: 175px;height: 174px;display: block;margin:10px;background:url(a-qp.png) no-repeat;}
</style>
<div class="img1 img" data-item="4" data-timer="100" data-height="236"></div>
<div class="img2 img" data-item="7" data-timer="120" data-height="174"></div>

JS-调用


//帧动画播放（仅限于背景图片，为兼容IE7及以上版本而存在）
/**
* 此处会返回方法，方便与更好的控制
* @param {[height]} 单个元素的高度（若为字符串格式，则在元素内进行查找）
* @param {[act]} 图片播放总个数（若为字符串格式，则在元素内进行查找）
* @param {[time]} 播放时间（若为字符串格式，则在元素内进行查找）
* @param {[num]} 播放次数，默认为循环
*
* @return {[timer]} 返回当前是否正在执行
* @return {[start]} 返回正在执行的方法
* @return {[stop]} 返回停止执行的方法
*/
var callback = $(".img").bgAnimation({
height: 'height',
act: 'item',
time: 'timer'
});

$(".img").click(function(){
var index = $(this).index();
callback[index].timer ? callback[index].stop() : callback[index].start();
});

// 下拉菜单
$(".form-select-text").click(function(){
var _this = $(this);
var iUl = _this.next("ul");
if (iUl.css("visibility")=="hidden") {
iUl.addClass("select-active");
} else {
iUl.removeClass("select-active");
}
});
$(".form-select ul li").click(function(){
var id = $(this).parents(".form-select").data("id");
var text = $(this).text();
$("#"+id).val(text);
//console.log($("#"+id).val());
$(".form-select[data-id="+id+"]").find("p").html(text);
$(".form-select[data-id="+id+"]").find("ul").removeClass("select-active");
});
//checkbox模拟
$(".form-checkbox ul li").click(function(){
var id = $(this).parents(".form-checkbox").data("id");
if($(this).hasClass("checked")){
$(this).removeClass("checked");
}else{
$(this).addClass("checked");
}
//checkbox传值
checkboxVal(id);
});
function checkboxVal(id){
var _arr = [];
$(".form-checkbox[data-id="+id+"] ul li").each(function(){
if($(this).hasClass("checked")){
_arr.push(1);
}else{
_arr.push(0);
}
});
var _text = _arr.join(":");
$("#"+id).val(_text);
// console.log($("#"+id).val());
}
//单选radio
$(".form-radio ul li").click(function(){
var index = $(this).index();
var id = $(this).parents(".form-radio").data("id");
$(this).addClass("checked").siblings("li").removeClass("checked");
$("#"+id).val(index);
// console.log($("#"+id).val());
});

判断当前浏览器版本 移动端和IE6的判断解决方法

//判断是否是IE6
var isIE6 = /msie 6/i.test(navigator.userAgent);
//判断是否是IE
var isIE = /msie/i.test(navigator.userAgent);
//返回版本值
var browser = (function(){
var isIE6 = /msie 6/i.test(navigator.userAgent);
var isIE7 = /msie 7/i.test(navigator.userAgent);
var isIE8 = /msie 8/i.test(navigator.userAgent);
var isIE = /msie/i.test(navigator.userAgent);
return {
msie:isIE,
version:function(){
switch(true){
case isIE6:return 6;
case isIE7:return 7;
case isIE8:return 8;
}
}()
};
})();
alert(browser.msie);
alert(browser.version);
